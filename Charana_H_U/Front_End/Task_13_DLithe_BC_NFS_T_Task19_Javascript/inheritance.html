JavaScript Inheritance

To create a class inheritance, use the extends keyword.
A class created with a class inheritance inherits all the methods from another class

Example:

            
                class Car {
                    constructor(brand) {
                        this.carname = brand;
                    }
                    present() {
                        return 'I have a ' + this.carname;
                    }
                }

                class Model extends Car {
                    constructor(brand, mod) {
                        super(brand);
                        this.model = mod;
                    }
                    show() {
                        return this.present() + ', it is a ' + this.model;
                    }
                }

                let myCar = new Model("Ford", "Mustang");
                document.getElementById("demo").innerHTML = myCar.show();

Getter and Setter

Getter and Setter are used to control access to an object's properties.

Example:

            
                class Employee {
                    constructor(name, id) {
                        this.name = name;
                        this.id = id;
                    }
                    get employeeName() {            //values get displayed(read)
                        return this.name;
                    }
                    set employeeName(name) {        //values get inititized(write)
                        this.name = name;
                    }
                }

                let employee = new Employee("John", 1);
                document.getElementById("demo1").innerHTML = employee.employeeName;

                employee.employeeName = "Pete";
                document.getElementById("demo1").innerHTML = employee.employeeName;

JavaScript Overloading

Overloading is a feature of JavaScript that allows you to define multiple functions with the same name.

Example:

            
                class display {
                    disp(x, y) {
                        document.write("2 arguments" + "
" + x + "
" + y);
                    }
                    disp(x, y, z) {
                        document.write("3 Arguments"+ "
" + x + "
" + y + "
" + z);
                    }
                }
                var d = new display();
                d.disp(4, 3);
                d.disp(4, 3, 2);

JavaScript Overriding

Overriding is a feature of JavaScript that allows you to define a function with the same name as a parent class's function.

Example:

            
                class Shape {
                    constructor(name) {
                        this.name = name;
                    }
                    getName() {
                        return this.name;
                    }
                }

                class Circle extends Shape {
                    constructor(name, radius) {
                        super(name);
                        this.radius = radius;
                    }
                    getName() {
                        return super.getName() + " " + this.radius;
                    }
                }

                let circle = new Circle("Circle", 5);
                document.getElementById("demo4").innerHTML = circle.getName();
